REPLACE the getExpirationStatus function with this logic:

export function getExpirationStatus(expirationDate: string, assignedAt: string): ExpirationStatus {
  const now = new Date();
  const expDate = new Date(expirationDate);

  if (isNaN(expDate.getTime())) {
    return 'expired';
  }

  // Strip time components — compare dates only
  const todayMidnight = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  const expMidnight = new Date(expDate.getFullYear(), expDate.getMonth(), expDate.getDate());
  const diffMs = expMidnight.getTime() - todayMidnight.getTime();
  const diffDays = Math.round(diffMs / (1000 * 60 * 60 * 24));

  // Future (expiration is tomorrow or later) = green
  if (diffDays > 0) return 'fresh';

  // Within last 3 days inclusive of today (diffDays is 0, -1, or -2) = yellow/warning
  if (diffDays >= -2) return 'warning';

  // More than 3 days in the past (diffDays <= -3) = red/expired
  return 'expired';
}
